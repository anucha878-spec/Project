1. ตรวจสอบ ms_ri_treaty.sp_bk_flag ที่ adw-etl <![CDATA[select sp_bk_flag from ms_ri_treaty where treaty_code = :treaty_code;]]>
2. sum ข้อมูลจาก tx_ri_std_all ด้วย field ต่อไปนี้ไปที่ total_premium ตามเงื่อนไข

- if ms_ri_treaty.sp_bk_flag = 'A' > sum(premium_life,premium_extra_life,premium_rider,premium_extra_rider, premium_add,premium_add_extra, premium_life_fac, premium_extra_life_fac, premium_rider_fac, premium_extra_rider_fac)if ms_ri_treaty.sp_bk_flag = 'B' > sum(premium_life,premium_extra_life,premium_add,premium_add_extra,premium_ttd,premium_extra_ttd,premium_tpd,premium_extra_tpd)if ms_ri_treaty.sp_bk_flag = 'C' > sum(premium_life,premium_extra_life)if ms_ri_treaty.sp_bk_flag = 'D' > sum(premium_add,premium_add_extra)if ms_ri_treaty.sp_bk_flag = 'E' > sum(premium_rider,premium_extra_rider)if ms_ri_treaty.sp_bk_flag = 'H' > sum(premium_life,premium_extra_life,premium_add,premium_add_extra,premium_ttd,premium_extra_ttd,premium_tpd,premium_extra_tpd,premium_rider,premium_extra_rider)
