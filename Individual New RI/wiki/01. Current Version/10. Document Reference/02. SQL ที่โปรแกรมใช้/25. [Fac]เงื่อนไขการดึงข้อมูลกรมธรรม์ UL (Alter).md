0. count ข้อมูลจากตาราง landing tx_ri_ul_alteration_dt สำหรับกรองข้อมูล Alter ที่เข้าเงื่อนไข alteration_code ,alteration_date ตรงตามงวด
1. หา alterId จากตาราง landing tx_ri_ul_alteration_dt สำหรับกรองข้อมูล Alter ที่เข้าเงื่อนไข alteration_code ,alteration_date ตรงตามงวด
2. นำ alterId ที่ได้มาหาต่อ (ค้นหาข้อมูลที่ เคยส่ง new renew กรณี Base)
3.นำ alterId ที่ได้มาหาต่อ (ค้นหาข้อมูลที่ เคยส่ง new renew กรณี Rider)
with landing as (   select   alt.policy_no,  alt.plan_code as plan_code ,  '0'::text as rider_code ,  alt.commencement_date,   alt.alteration_date ,  null::date as alteration_effective_date ,   alt.alteration_date as alter_eff_date_con,   null::date as created_date_alter,   alt.update_date as updated_date_alter,  null::date as alteration_update_date,   null::date as lapse_date,   alt.maturity_date ,  alt.sum_insured_life as initial_sa ,  alt.previous_sa as previous_initial_sa,  null::numeric as premium_base,   null::numeric as premium_extra_base,   alt.remark as alteration_code ,  null::text as alteration_case,   alt.alteration_date as alter_eff_date,   null::text as previous_policy_status,   null::numeric as percent_emr,   clc2.description as policy_status_desc_alt   ,fac.effective_date_from as effective_date_from_fac   ,fac.effective_date_to as effective_date_to_fac  from tx_ri_ul_alteration_dt alt  inner join cf_policy_fac_hd fac on fac.policy_no = alt.policy_no   LEFT JOIN cf_lookup_catalog clc2 on clc2.parent_id = 1004400 and clc2.lookup_key = alt.remark  where alt.ri_ul_alteration_id = :altId  order by fac.cf_policy_fac_hd_id desc limit 1   ),  policy_dt as (  select pol.ri_est_policy_dt_id, pol.policy_no,  pol.policy_status ,  pol.policy_status_desc,  pol.policy_type,  pol.coverage_from,  pol.coverage_to,  pol.std_substd as standard,  pol.sale_channel_code,  pol.cession_no,  pol.approved_case,  pol.cert_no,  pol.policy_term,  pol.plan_type, pol.std_fac,  pol.con_age as cust_issued_age,  pol.occ_class as cust_occ_class,  pol.cust_occ_name as cust_occ_name,  pol.commencement_date as commencement_date_est_act,  pol.coverage_from as current_effective_date_from,  pol.coverage_to as current_effective_date_to ,  cust.cust_cis_no as cust_cis_no,  cust.id_no as cust_id_card ,  cust.title as cust_title_name ,  cust.name as cust_name ,  cust.surname as cust_surname ,  cust.gender as cust_sex,  cust.birth_date as cust_birth_date,  cust.area as province_eng,  cust.country as country_eng,  pol.plan_id ,  pol.remark ,  hd.period,  pol.sale_channel  from tx_ri_est_hd hd  inner join tx_ri_est_policy_dt pol on pol.ri_est_hd_id = hd.ri_est_hd_id  inner join landing l on l.policy_no = pol.policy_no  left join tx_ri_est_cust_dt cust on cust.ri_est_cust_dt_id = pol.ri_est_cust_dt_id  where pol.ri_est_policy_dt_id = ( SELECT ri_est_policy_dt_id FROM tx_ri_est_base_auto_dt WHERE ri_est_base_auto_dt_id = :riEstRiderDtId )  and  CASE   WHEN (pol.policy_type = 'ORD' and l.alteration_code in ('IN', 'AP', 'EX', 'RP')) ") ;   or (pol.policy_type = 'ORD' and l.alteration_code = 'RE' and l.previous_initial_sa > l.initial_sa)    or (pol.policy_type = 'IND' and l.alteration_code = '6')    or (pol.policy_type = 'UL' and l.alteration_code in ('Increase SA', 'Decrease SA')) //แบบ A.1 , แบบ E.1 //ตรวจสอบเพิ่มเติมกรณี tx_ri_ord_alteration_dt.alteration_code = IN , EX , RP , RE   THEN  NOT EXISTS (    SELECT 1 FROM tx_ri_send_reinsurer_period sub    WHERE hd.treaty_code = sub.treaty_code and pol.policy_no = sub.policy_no and :period = sub.period    ) and l.alter_eff_date_con between pol.coverage_from and pol.coverage_to   WHEN (pol.policy_type = 'ORD' and l.alteration_code in ('CC', 'CD', 'LP', 'SD'))    or (pol.policy_type = 'PA' and l.alteration_code in ('C', 'L', 'Z'))    or (pol.policy_type = 'IND' and l.alteration_code = '7')    or (pol.policy_type = 'UL' and l.alteration_code in ('4', '6', '8', '9')) then l.alter_eff_date_con <= pol.coverage_to //แบบ B คืน Advance ทั้งหมด  WHEN (pol.policy_type = 'ORD' and l.alteration_code = 'DE')    or (pol.policy_type = 'UL' and l.alteration_code = '11') THEN true //แบบ C คืนทุกรอบที่เคยส่ง //ตรวจสอบเพิ่มเติมกรณี tx_ri_ord_alteration_dt.alteration_code = DE  WHEN pol.policy_type = 'ORD' and l.alteration_code = 'RE' and l.previous_initial_sa = l.initial_sa THEN l.commencement_date = pol.commencement_date //แบบ D ทำตามรอบที่เกิดขึ้นของวันที่เริ่มต้นสัญญา Previous Initial SA = Initial SA = RE2  WHEN pol.policy_type = 'PA' and l.alteration_code = 'B' THEN l.alter_eff_date_con between pol.coverage_from and pol.coverage_to //แบบ F - คืน Refund ในรอบนั้น  ELSE l.alter_eff_date_con between pol.coverage_from and pol.coverage_to   END  )  select  alt.policy_no ,    alt.plan_code ,    alt.rider_code,    alt.alteration_date ,    alt.alteration_effective_date,    alt.created_date_alter ,    alt.updated_date_alter ,    alt.alteration_update_date ,    alt.lapse_date ,    alt.maturity_date ,    alt.initial_sa ,    alt.previous_initial_sa ,    alt.premium_base ,  alt.premium_extra_base ,    alt.alteration_code ,    alt.alteration_case ,    alt.previous_policy_status,    po.policy_status ,    alt.policy_status_desc_alt as policy_status_desc,    po.policy_type,    alt.commencement_date,    po.sale_channel_code,    po.cession_no,    po.approved_case,    po.std_fac,    po.cert_no,    po.cust_cis_no,    po.cust_id_card ,    po.cust_title_name ,    po.cust_name ,    po.cust_surname ,    po.cust_sex,    po.cust_issued_age,    po.cust_birth_date,    po.cust_occ_class,    po.cust_occ_name,    po.province_eng,    po.country_eng,    po.standard,    alt.percent_emr,    po.policy_term,    po.plan_type,   po.commencement_date_est_act,  po.current_effective_date_from ,  po.current_effective_date_to,  po.plan_id ,  po.remark ,  po.sale_channel,   po.period,     base.coverage_type ,  base.actual_sa_life as param_actual_sa_life,  base.previous_sa_life as param_previous_actual_sa_life,  base.reserve_amount_life as param_reserve_amount_life,  base.ri_current_nar_life as param_nar_life,  base.ri_previous_nar_life as param_previous_nar_life,  base.total_nar_life as param_total_nar_life,  base.ri_sum_assured_life as param_sr_life,  base.previous_sr_life as param_previous_sr_life,  base.total_sr_life as param_total_sr_life,  base.ri_ratio_life as param_ri_ratio_life,  base.premium_life as param_ri_premium_life,  base.premium_extra_life as param_ri_ext_premium_life,  base.com_rate,  base.comm_life as param_ri_comm_life,  base.comm_extra_life as param_ri_ext_comm_life,  base.emr_life,  base.net_ri_premium as param_net_ri_premium  from tx_ri_est_base_auto_dt base  inner join policy_dt po on base.ri_est_policy_dt_id = po.ri_est_policy_dt_id  left join landing alt on po.policy_no = alt.policy_no  where base.code_plan = alt.plan_code and base.report_type in ('NEW', 'RENEW') and base.ri_est_base_auto_dt_id = :riEstRiderDtId
